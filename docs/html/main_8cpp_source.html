98void parsearLinea(const char *linea, char *tipo, char *id, char *valor)
99{
100 char lineaCopia[256];
101 std::strncpy(lineaCopia, linea, 255);
102 lineaCopia[255] = '\0';
103
104 char *token = std::strtok(lineaCopia, ":");
105 if (token != nullptr)
106 {
107 std::strcpy(tipo, token);

109
110 token = std::strtok(nullptr, ":");
111 if (token != nullptr)
112 {
113 std::strcpy(id, token);
